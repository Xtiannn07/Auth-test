rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // ==================== USERNAMES COLLECTION ====================
    match /usernames/{username} {
      allow get: if true;
      allow create, update, delete: if request.auth != null && 
                                   request.auth.uid == request.resource.data.uid;
    }

    // ==================== USERS COLLECTION ====================
    match /users/{userId} {
      allow read: if true;
      allow create: if request.auth != null && 
                   request.auth.uid == userId;
      // Allow users to update their own profile
      allow update: if request.auth != null && 
                    request.auth.uid == userId &&
                    onlyUpdatesAllowedFields(request.resource.data, resource.data);
      // Allow users to update followers/following lists for follow functionality
      allow update: if request.auth != null && 
                    onlyUpdatesFollowData(request.resource.data, resource.data) &&
                    isValidFollowOperation(request.resource.data, resource.data, request.auth.uid);
      
      // User's posts subcollection
      match /posts/{postId} {
        allow read: if true;
        allow create: if request.auth != null && 
                      request.auth.uid == userId &&
                      validPostDocument(request.resource.data);
      }
    }

    // ==================== FOLLOWING COLLECTION ====================
    match /following/{userId} {
      allow read: if true;
      allow create, update: if request.auth != null && request.auth.uid == userId;
    }

    // ==================== FOLLOWERS COLLECTION ====================
    match /followers/{userId} {
      allow read: if true;
      // Allow the user to manage their followers list
      allow create, update: if request.auth != null;
    }

    // ==================== POSTS COLLECTION ====================
    match /posts/{postId} {
      allow get, list: if true;
      allow create: if request.auth != null && 
                    validPostDocument(request.resource.data) &&
                    request.resource.data.author.id == request.auth.uid;
      allow update: if isPostAuthor() && 
                   withinEditWindow() && 
                   onlyUpdatesAllowedPostFields(request.resource.data, resource.data);
      allow update: if request.auth != null &&
                   onlyUpdatesLikes(request.resource.data, resource.data) &&
                   validLikeOperation(request.resource.data.likes, resource.data.likes);
      allow delete: if isPostAuthor();
    }

    // ==================== HELPER FUNCTIONS ====================
    function onlyUpdatesAllowedFields(newData, oldData) {
      return newData.diff(oldData).affectedKeys().hasOnly([
        "displayName",
        "bio",
        "website",
        "avatarUrl",
        "followers",
        "following",
        "followerCount",
        "followingCount"
      ]);
    }

    function isPostAuthor() {
      return resource != null && 
             resource.data != null &&
             resource.data.author != null &&
             resource.data.author.id != null &&
             request != null &&
             request.auth != null && 
             request.auth.uid == resource.data.author.id;
    }

    function withinEditWindow() {
      return request.time < resource.data.createdAt + duration.value(24, 'h');
    }

    function validPostDocument(data) {
      return data.keys().hasOnly([
        "title",
        "content",
        "author",
        "likes",
        "createdAt"
      ]) && data.title is string &&
           data.content is string &&
           data.author is map &&
           data.author.id is string &&
           data.author.name is string &&
           data.likes is list &&
           data.createdAt is timestamp;
    }

    function onlyUpdatesAllowedPostFields(newData, oldData) {
      return newData.diff(oldData).affectedKeys().hasOnly(["title", "content"]);
    }

    function onlyUpdatesLikes(newData, oldData) {
      return newData.diff(oldData).affectedKeys().hasOnly(["likes"]);
    }

    function validLikeOperation(newLikes, oldLikes) {
      // Ensure only one like change at a time
      return newLikes.size() == oldLikes.size() + 1 ||
             newLikes.size() == oldLikes.size() - 1;
    }

    // Function to check if only follower/following data is being modified
    function onlyUpdatesFollowData(newData, oldData) {
      return newData.diff(oldData).affectedKeys().hasOnly([
        "followers",
        "following",
        "followerCount",
        "followingCount"
      ]);
    }
    
    // Function to validate that the follow operation is legitimate
    function isValidFollowOperation(newData, oldData, authUid) {
      // Case 1: User is updating their own document to add/remove someone from their following list
      let isUpdatingOwnDocument = request.auth.uid == resource.id;
      let isUpdatingFollowing = newData.following != null && oldData.following != null && 
                              (newData.diff(oldData).affectedKeys().hasOnly(['following', 'followingCount']));
      
      // Case 2: User is updating someone else's document to add/remove themselves from followers list
      let isUpdatingSomeoneElsesDocument = request.auth.uid != resource.id;
      let hasFollowersData = newData.followers != null && oldData.followers != null;
      let isUpdatingFollowers = hasFollowersData && 
                              (newData.diff(oldData).affectedKeys().hasOnly(['followers', 'followerCount']));
      let isAddingOrRemovingSelf = (
        // Adding follower
        (newData.followers[authUid] != null && oldData.followers[authUid] == null) ||
        // Removing follower
        (newData.followers[authUid] == null && oldData.followers[authUid] != null)
      );
      
      // Return true if either case is valid
      return (isUpdatingOwnDocument && isUpdatingFollowing) || 
             (isUpdatingSomeoneElsesDocument && isUpdatingFollowers && isAddingOrRemovingSelf);
    }

    // ==================== DEFAULT DENY ====================
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
